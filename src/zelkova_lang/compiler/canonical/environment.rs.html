17/// Environment represent the set of values/types available to a compilation unit.
18/// It is derived from previous successful units and is provisioned through a
19/// module imports.
20///
21/// As a whole, the aim of the canonical AST is to not have to worry about the
22/// Environment in later phases. We still need one to translate the parser AST.
23/// The good news being, it can be local to the canonicalization function.
24pub trait Environment<'parent>: std::fmt::Debug {
25    fn find_type(&self, name: &Name) -> Option<&Type>;
26
27    fn module_name(&self) -> &ModuleName;
28
29    fn find_value(&self, name: &Name) -> Option<&ValueType>;
30
31    fn find_type_constructor(&self, name: &Name) -> Option<&TypeConstructor>;
32
33    fn local_infix_exists(&self, name: &Name) -> bool;
34
35    fn insert_local_value(&mut self, name: &Name);
36
37    // 'parent must lives at least as long as 'a
38    fn new_scope<'a>(&'a self) -> ScopedEnvironment<'parent, 'a>
39    where
40        'parent: 'a;
41}

43pub fn new_environment(
44    module_name: &ModuleName,
45    interfaces: &HashMap<Name, Interface>,
46    imports: &Vec<parser::Import>,
47) -> Result<RootEnvironment, Vec<EnvError>> {
48    let mut env = RootEnvironment {
49        module_name: module_name.clone(),
50        infixes: HashMap::new(),
51        types: HashMap::new(),
52        constructors: HashMap::new(),
53        variables: HashMap::new(),
54    };
55    let mut errors = vec![];
56
57    for parser::Import {
58        name,
59        alias,
60        exposing,
61    } in imports
62    {
63        match process_import(&mut env, interfaces, &name, &alias, &exposing) {
64            Ok(_) => (),
65            Err(err) => {
66                errors.push(err);
67            }
68        }
69    }
70
71    if errors.is_empty() {
72        Ok(env)
73    } else {
74        Err(errors)
75    }
76}

78fn process_import(
79    env: &mut RootEnvironment,
80    interfaces: &HashMap<Name, Interface>,
81    imported_module_name: &Name,
82    alias: &Option<Name>,
83    exposing: &parser::Exposing,
84) -> Result<(), EnvError> {
85    let interface = interfaces
86        .get(&imported_module_name)
87        .ok_or_else(|| EnvError::InterfaceNotFound(imported_module_name.clone()))?;
88
89    trace!(
90        "process_import(imported_module_name={:?}, alias={:?}, exposing: {:?})",
91        imported_module_name,
92        alias,
93        exposing
94    );
95
96    // First we insert all values/types from the module, prefixed with the module name or its alias
97    let prefix = alias.as_ref().unwrap_or(imported_module_name);
98
99    for (value_name, tpe) in &interface.values {
100        insert_foreign_value(
101            env,
102            value_name.qualify_with_name(prefix).unwrap().to_name(),
103            tpe.clone(),
104            &interface.module_name,
105        );
106    }
107
108    for (union_name, union) in &interface.unions {
109        insert_foreign_union_type(env, Some(prefix), union_name, union.variants.iter());
110    }
111
112    // TODO Infix ?
113
114    // Then we insert unqualified values/types
115    match exposing {
116        parser::Exposing::Open => {
117            // We add everything to the current environment
118
119            for (value_name, tpe) in &interface.values {
120                insert_foreign_value(env, value_name.clone(), tpe.clone(), &interface.module_name);
121            }
122
123            for (op_name, infix) in &interface.infixes {
124                env.infixes.insert(op_name.clone(), infix.clone());
125            }
126
127            // We need to insert the type without any qualifier, including variants
128            for (union_name, union) in &interface.unions {
129                insert_foreign_union_type(env, None, union_name, union.variants.iter());
130            }
131        }
132
133        parser::Exposing::Explicit(exposeds) => {
134            // We only add the explicitly named types/values/infixes
135            let iter = exposeds.iter().map(|exposed| {
136                match exposed {
137                    parser::Exposed::Lower(value_name) => {
138                        let tpe = interface
139                            .values
140                            .get(&value_name)
141                            .ok_or_else(|| EnvError::ValueNotFound(value_name.clone()))?;
142
143                        insert_foreign_value(
144                            env,
145                            value_name.clone(),
146                            tpe.clone(),
147                            &interface.module_name,
148                        );
149                    }
150                    parser::Exposed::Upper(type_name, parser::Privacy::Private) => {
151                        let tpe = Type::Type(type_name.clone(), vec![]);
152
153                        // Add the type without qualifier and without constructors (they are private)
154                        env.types.insert(type_name.clone(), tpe);
155                    }
156                    parser::Exposed::Upper(type_name, parser::Privacy::Public) => {
157                        let union = interface
158                            .unions
159                            .get(&type_name)
160                            .ok_or_else(|| EnvError::UnionNotFound(type_name.clone()))?;
161
162                        insert_foreign_union_type(env, None, type_name, union.variants.iter());
163                    }
164                    parser::Exposed::Operator(variable_name) => {
165                        let infix = interface
166                            .infixes
167                            .get(&variable_name)
168                            .ok_or_else(|| EnvError::InfixNotFound(variable_name.clone()))?;
169
170                        env.infixes.insert(variable_name.clone(), infix.clone());
171                        // How do we represent infixes ?
172                        // When do we do rewrite them ?
173                    }
174                };
175
176                Ok(())
177            });
178
179            collect_accumulate(iter).map_err(EnvError::Multiple)?;
180        }
181    };
182
183    Ok(())
184}

186fn insert_foreign_union_type<'a, I: Iterator<Item = &'a TypeConstructor>>(
187    env: &mut RootEnvironment,
188    qualifier: Option<&Name>,
189    union_name: &Name,
190    variants: I,
191) {
192    // If there is a given qualifier use it, otherwise use the name as is
193    let qualify = |n: &Name| {
194        qualifier
195            .and_then(|q| n.qualify_with_name(q))
196            .map(|q| q.to_name())
197            .unwrap_or(n.clone())
198    };
199
200    env.types
201        .insert(qualify(union_name), Type::Type(union_name.clone(), vec![]));
202
203    for variant in variants {
204        // Variant are not qualified, which means we have to alias/qualify them as needed
205        env.constructors
206            .insert(qualify(&variant.name), variant.clone());
207    }
208}

210fn insert_foreign_value(
211    env: &mut RootEnvironment,
212    name: Name,
213    tpe: Type,
214    module_name: &ModuleName,
215) {
216    let vt = ValueType::Foreign(module_name.clone(), tpe.clone());
217
218    // Can it be done more efficiently by using get_mut ?
219    match env.variables.remove(&name) {
220        Some(ValueType::Foreign(module, _)) => {
221            env.variables
222                .insert(name, ValueType::Foreigns(vec![module_name.clone(), module]));
223        }
224        Some(ValueType::Foreigns(mut vec)) => {
225            vec.push(module_name.clone());
226            env.variables.insert(name, ValueType::Foreigns(vec));
227        }
228        None => {
229            env.variables.insert(name, vt);
230        }
231        _ => todo!("find out what to do in those cases"),
232    }
233}

244/// RootEnvironment represents the top level module and contains information accessible
245/// from the entire module.
246///
247/// This is opposed to a ScopedEnvironment which contains
248/// additional information available only to a scoped expression (eg. local variable)
249#[derive(Debug)]
250pub struct RootEnvironment {
251    module_name: ModuleName,
252    infixes: HashMap<Name, Infix>,
253    types: HashMap<Name, Type>,
254    constructors: HashMap<Name, TypeConstructor>,
255    variables: HashMap<Name, ValueType>,
256}
257
258impl RootEnvironment {
259    // TODO Do we need a local/foreign distinction for infixes ? (or in general ?)
260    pub fn insert_local_infix(&mut self, name: Name, infix: Infix) {
261        self.infixes.insert(name, infix);
262    }
263
264    // TODO Use insert_foreign_value (and rename to remove the foreign part)
265    // TODO Return an error if declaration already exists
266    pub fn insert_top_level_value(&mut self, name: Name) {
267        self.variables.insert(name, ValueType::TopLevel);
268    }
269
270    // TODO Use insert_foreign_union_type (and rename to remove the foreign part)
271    pub fn insert_union_type(&mut self, name: Name, union: UnionType) {
272        let args = union
273            .variables
274            .iter()
275            .map(|t| Type::Variable(t.clone()))
276            .collect();
277        let tpe = Type::Type(name.clone(), args);
278        self.types.insert(name.clone(), tpe);
279
280        for tctor in union.variants {
281            self.constructors.insert(tctor.name.clone(), tctor.clone());
282        }
283    }
284}

330/// An Environment scoped to a module's sub expression (`let`, function, etc…)
331#[derive(Debug)]
332pub struct ScopedEnvironment<'root, 'parent> {
333    parent: &'parent dyn Environment<'root>,
334    variables: HashMap<Name, ValueType>,
335}
336
337impl<'root, 'parent> Environment<'parent> for ScopedEnvironment<'root, 'parent> {
338    fn find_type(&self, name: &Name) -> Option<&Type> {
339        self.parent.find_type(name)
340    }
341
342    fn module_name(&self) -> &ModuleName {
343        &self.parent.module_name()
344    }
345
346    fn find_value(&self, name: &Name) -> Option<&ValueType> {
347        self.variables.get(name).or(self.parent.find_value(name))
348    }
349
350    fn find_type_constructor(&self, name: &Name) -> Option<&TypeConstructor> {
351        self.parent.find_type_constructor(name)
352    }
353
354    fn local_infix_exists(&self, name: &Name) -> bool {
355        self.parent.local_infix_exists(&name)
356    }
357
358    // TODO Return error if name already exists
359    // Needs to reuse the logic in import_foreign_value
360    fn insert_local_value(&mut self, name: &Name) {
361        self.variables.insert(name.clone(), ValueType::Local);
362    }
363
364    fn new_scope<'a>(&'a self) -> ScopedEnvironment<'parent, 'a>
365    where
366        'parent: 'a,
367    {
368        let parent: &'a dyn Environment<'parent> = self;
369
370        ScopedEnvironment {
371            parent,
372            variables: HashMap::new(),
373        }
374    }
375}

377impl<'root, 'parent> ScopedEnvironment<'root, 'parent> {
378    pub fn expose_pattern(&mut self, pattern: &Pattern) {
379        match pattern {
380            Pattern::Anything => (),
381            Pattern::Int(_) => (),
382            Pattern::Float(_) => (),
383            Pattern::Char(_) => (),
384            Pattern::Bool(_) => (),
385
386            Pattern::Variable(n) => {
387                self.variables.insert(n.clone(), ValueType::Local);
388            }
389            Pattern::Tuple(a, b, c) => {
390                self.expose_pattern(a);
391                self.expose_pattern(b);
392                if let Some(c) = c {
393                    self.expose_pattern(c);
394                }
395            }
396            Pattern::Constructor { args, .. } => {
397                for arg in args {
398                    self.expose_pattern(arg);
399                }
400            }
401        }
402    }
403}
