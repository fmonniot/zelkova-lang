6/// Returns true if `tpe` is a numeric type (Int, Float, or Number).
7fn is_numeric(tpe: &Type) -> bool {
8    matches!(
9        tpe,
10        Type::Literal(TypeLiteral::Int) | Type::Literal(TypeLiteral::Float) | Type::Number
11    )
12}

14pub(super) fn unify(constraints: HashSet<Constraint>) -> Result<Substitution, Error> {
15    debug!("unify: {:?}", constraints);
16    let mut iter = constraints.iter();
17
18    match iter.next() {
19        None => Ok(Substitution::empty()),
20        Some(first) => {
21            let sub_head = unify_one_constraint(first)?;
22
23            // Apply this substitution to the remaining constraints
24            let constraints_tail: HashSet<_> = iter.map(|c| sub_head.apply(c)).collect();
25
26            // Then recursively unify the substituted constraints
27            let sub_tail = unify(constraints_tail)?;
28
29            // And finally merged the unified substitution with the first one
30            Ok(sub_head.merge(sub_tail))
31        }
32    }
33}

35fn unify_one_constraint(Constraint(a, b): &Constraint) -> Result<Substitution, Error> {
36    debug!("unify_one_constraint: {:?} to {:?}", a, b);
37    match (a, b) {
38        (Type::Literal(TypeLiteral::Bool), Type::Literal(TypeLiteral::Bool)) => {
39            Ok(Substitution::empty())
40        }
41        (Type::Literal(TypeLiteral::Int), Type::Literal(TypeLiteral::Int)) => {
42            Ok(Substitution::empty())
43        }
44        (Type::Literal(TypeLiteral::Char), Type::Literal(TypeLiteral::Char)) => {
45            Ok(Substitution::empty())
46        }
47        (Type::Literal(TypeLiteral::Float), Type::Literal(TypeLiteral::Float)) => {
48            Ok(Substitution::empty())
49        }
50        (
51            Type::Fun {
52                param_tpe: p1,
53                return_tpe: r1,
54            },
55            Type::Fun {
56                param_tpe: p2,
57                return_tpe: r2,
58            },
59        ) => {
60            let mut constraints = HashSet::new();
61
62            constraints.insert(Constraint(*p1.clone(), *p2.clone()));
63            constraints.insert(Constraint(*r1.clone(), *r2.clone()));
64
65            unify(constraints)
66        }
67        // Number unifies with Int, Float, or another Number (but not Bool, Char, etc.)
68        (Type::Number, other) | (other, Type::Number) if is_numeric(other) => {
69            Ok(Substitution::empty())
70        }
71        // Tuples: unify element-by-element (must have matching arity)
72        (Type::Tuple(a1, b1, None), Type::Tuple(a2, b2, None)) => {
73            let mut cs = HashSet::new();
74            cs.insert(Constraint(*a1.clone(), *a2.clone()));
75            cs.insert(Constraint(*b1.clone(), *b2.clone()));
76            unify(cs)
77        }
78        (Type::Tuple(a1, b1, Some(c1)), Type::Tuple(a2, b2, Some(c2))) => {
79            let mut cs = HashSet::new();
80            cs.insert(Constraint(*a1.clone(), *a2.clone()));
81            cs.insert(Constraint(*b1.clone(), *b2.clone()));
82            cs.insert(Constraint(*c1.clone(), *c2.clone()));
83            unify(cs)
84        }
85        // ADT types: must have same name and same number of args; unify args pairwise
86        (Type::Adt(n1, args1), Type::Adt(n2, args2)) if n1 == n2 && args1.len() == args2.len() => {
87            let constraints: std::collections::HashSet<_> = args1
88                .iter()
89                .zip(args2.iter())
90                .map(|(a, b)| Constraint(a.clone(), b.clone()))
91                .collect();
92            unify(constraints)
93        }
94        (Type::Variable(tvar), tpe) => unify_variable(tvar, tpe),
95        (tpe, Type::Variable(tvar)) => unify_variable(tvar, tpe),
96        (a, b) => Err(Error::UnificationFailed {
97            left: a.clone(),
98            right: b.clone(),
99        }),
100    }
101}

103fn unify_variable(tvar: &TypeVariable, tpe: &Type) -> Result<Substitution, Error> {
104    match tpe {
105        Type::Variable(tvar2) => {
106            if tvar == tvar2 {
107                Ok(Substitution::empty())
108            } else {
109                Ok(Substitution::one(tvar.clone(), tpe.clone()))
110            }
111        }
112        _ => {
113            if occurs(tvar, tpe) {
114                Err(Error::CircularType)
115            } else {
116                Ok(Substitution::one(tvar.clone(), tpe.clone()))
117            }
118        }
119    }
120}
121
122fn occurs(tvar: &TypeVariable, tpe: &Type) -> bool {
123    match tpe {
124        Type::Fun {
125            param_tpe,
126            return_tpe,
127        } => occurs(tvar, param_tpe) || occurs(tvar, return_tpe),
128        Type::Tuple(a, b, c) => {
129            occurs(tvar, a) || occurs(tvar, b) || c.as_ref().is_some_and(|t| occurs(tvar, t))
130        }
131        Type::Adt(_, args) => args.iter().any(|a| occurs(tvar, a)),
132        Type::Variable(tvar2) => tvar == tvar2,
133        _ => false,
134    }
135}
