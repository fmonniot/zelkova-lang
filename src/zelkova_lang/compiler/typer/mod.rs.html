1//! This module contains the type checker pass of the language
2//!
3//! It works with the source AST and will perform two jobs:
4//! - type checks the different declarations and expression
5//! - infer the types when not declared in the source
6//!
7//! I have no idea how that works; so bear with me while I explore
8//! the space, make mistake and (hopefully) learn something :)
9//!
10//! Some papers on type inference:
11//! - http://steshaw.org/hm/hindley-milner.pdf
12//! - https://pdfs.semanticscholar.org/8983/233b3dff2c5b94efb31235f62bddc22dc899.pdf
13//! - http://gallium.inria.fr/~fpottier/publis/fpottier-elaboration.pdf
14//! - http://gallium.inria.fr/~fpottier/publis/emlti-final.pdf
15//!
16//! A type inference problem consists of a type environment Γ , an expression t, and a type T of kind ?
17//!
18//! Constraint generation rules:
19//!
20//! - Equation 1: ⟦x : T⟧ = x ≼ T
21//!   "x has type T if and only if T is an instance of the type scheme associated with x"
22//!   Important part: There is no relation to the typing environment Γ, instead x appears free (and will be bound to Γ later)
23//!
24//! - Equation 2: ⟦λz.t : T⟧ = ∃X1X2.(let z : X1 in ⟦t : X2⟧ ∧ X1 → X2 ≤ T)
25//!   "λz.t has type T if and only if, for some X1 and X2,
26//!   (i) under the assumption that z has type X1, t has type X2, and
27//!   (ii) T is a supertype of X1 → X2."
28//!   z and t types must be fresh (can't generally guess them). They are _existentially_ bound because we are going to
29//!   solve their values. Note that z is _not_ fresh in the condition (i).
30//!
31//! - Equation 3: ⟦t1 t2 : T⟧ = ∃X2.(⟦t1 : X2 → T⟧ ∧ ⟦t2 : X2⟧)
32//!   "t1 t2 has type T if and only if, for some X2, t1 has type X2 → T and t2 has type X2"
33//!
34//! - Equation 4: ⟦let z = t1 in t2 : T⟧ = let z : ∀X[⟦t1 : X⟧].X in ⟦t2 : T⟧
35//!   "let z = t1 in t2 has type T if and only if, under the assumption that z has every type X such that ⟦t1 : X⟧ holds, t2 has type T"
36//!
37//!

44#[derive(Debug)]
45pub enum Error {
46    TypeMismatch { expected: Type, actual: Type },
47    UnificationFailed { left: Type, right: Type },
48    CircularType,
49    UnboundVariable(String),
50}

52pub fn type_check(module: &Module) -> Result<(), Error> {
53    // JavaScript binding modules use synthetic placeholder bodies — skip type checking.
54    if module.binding_javascript {
55        return Ok(());
56    }
57
58    // Start at a high offset to avoid collisions with the counter inside
59    // Types::new() (which starts at 10) used during inference.
60    let mut counter = 10_000u32;
61
62    // First pass: build global env from all TypedValues' declared types.
63    // This allows values to reference other module-level typed values.
64    let mut global: HashMap<String, Type> = HashMap::new();
65    for (name, value) in &module.values {
66        if let canonical::Value::TypedValue { tpe, .. } = value {
67            let mut var_map = HashMap::new();
68            if let Some(typer_tpe) = canonical_type_to_typer_type(tpe, &mut var_map, &mut counter) {
69                // Add both qualified (e.g. "Test.not") and unqualified (e.g. "not") names
70                let qname = module.name.qualify_name(name).to_name().0.clone();
71                global.insert(qname, typer_tpe.clone());
72                global.insert(name.0.clone(), typer_tpe);
73            }
74        }
75    }
76
77    // Second pass: add constructor types to global from module.types
78    for (type_name, union_type) in &module.types {
79        // Fresh type vars for each ADT type parameter (e.g. "a" in Maybe a)
80        let mut adt_var_map: HashMap<String, TypeVariable> = HashMap::new();
81        for tv_name in &union_type.variables {
82            counter += 1;
83            adt_var_map.insert(tv_name.0.clone(), TypeVariable { id: counter });
84        }
85
86        // Build result type: Adt(type_name, [TypeVar for each param])
87        let result_args: Vec<Type> = union_type
88            .variables
89            .iter()
90            .map(|v| Type::Variable(adt_var_map[&v.0].clone()))
91            .collect();
92        let result_type = Type::Adt(type_name.0.clone(), result_args);
93
94        for ctor in &union_type.variants {
95            let ctor_type = if ctor.type_parameters.is_empty() {
96                result_type.clone()
97            } else {
98                let mut translate_var_map = adt_var_map.clone();
99                let params: Vec<Type> = ctor
100                    .type_parameters
101                    .iter()
102                    .filter_map(|t| {
103                        canonical_type_to_typer_type(t, &mut translate_var_map, &mut counter)
104                    })
105                    .collect();
106                if params.len() != ctor.type_parameters.len() {
107                    continue; // untranslatable param type, skip this constructor
108                }
109                // Build Fun type: p1 -> p2 -> ... -> result_type
110                params
111                    .into_iter()
112                    .rev()
113                    .fold(result_type.clone(), |acc, p| Type::Fun {
114                        param_tpe: Box::new(p),
115                        return_tpe: Box::new(acc),
116                    })
117            };
118
119            // Register under both unqualified ("Just") and qualified ("Test.Just") names
120            global.insert(ctor.name.0.clone(), ctor_type.clone());
121            let qname = module.name.qualify_name(&ctor.name).to_name().0.clone();
122            global.insert(qname, ctor_type);
123        }
124    }
125
126    // Third pass: check each value
127    for value in module.values.values() {
128        let Some((term, annotation)) =
129            value_to_term_and_annotation(value, &module.types, &mut counter)
130        else {
131            continue; // unsupported construct — skip for now
132        };
133
134        let inferred = match infer(term, global.clone()) {
135            // An unbound variable means the term references a top-level function whose
136            // type we couldn't represent (e.g. Float-typed functions). Skip silently.
137            Err(Error::UnboundVariable(_)) => continue,
138            Err(e) => return Err(e),
139            Ok(t) => t,
140        };
141
142        // If there's a type annotation, verify inferred type is compatible
143        if let Some(ann) = annotation {
144            let mut constraints = std::collections::HashSet::new();
145            constraints.insert(Constraint(inferred.clone(), ann.clone()));
146            unifier::unify(constraints).map_err(|_| Error::TypeMismatch {
147                expected: ann,
148                actual: inferred,
149            })?;
150        }
151    }
152
153    Ok(())
154}

158/// Convert a canonical type to the typer's simplified Type representation.
159/// Returns None for types that cannot yet be represented (tuples, named types
160/// with parameters, etc.).
161///
162/// `var_map` maps named type variables (e.g. "a") to consistent TypeVariable
163/// ids, so that `a -> a` produces the same variable on both sides.
164fn canonical_type_to_typer_type(
165    tpe: &canonical::Type,
166    var_map: &mut HashMap<String, TypeVariable>,
167    counter: &mut u32,
168) -> Option<Type> {
169    match tpe {
170        canonical::Type::Type(name, args) if args.is_empty() && name.0 == "Int" => {
171            Some(Type::Literal(TypeLiteral::Int))
172        }
173        canonical::Type::Type(name, args) if args.is_empty() && name.0 == "Bool" => {
174            Some(Type::Literal(TypeLiteral::Bool))
175        }
176        canonical::Type::Type(name, args) if args.is_empty() && name.0 == "Char" => {
177            Some(Type::Literal(TypeLiteral::Char))
178        }
179        canonical::Type::Type(name, args) if args.is_empty() && name.0 == "Float" => {
180            Some(Type::Literal(TypeLiteral::Float))
181        }
182        canonical::Type::Variable(name) => {
183            let tv = var_map.entry(name.0.clone()).or_insert_with(|| {
184                *counter += 1;
185                TypeVariable { id: *counter }
186            });
187            Some(Type::Variable(tv.clone()))
188        }
189        canonical::Type::Arrow(a, b) => {
190            let a = canonical_type_to_typer_type(a, var_map, counter)?;
191            let b = canonical_type_to_typer_type(b, var_map, counter)?;
192            Some(Type::Fun {
193                param_tpe: Box::new(a),
194                return_tpe: Box::new(b),
195            })
196        }
197        canonical::Type::Tuple(a, b, c) => {
198            let a = canonical_type_to_typer_type(a, var_map, counter)?;
199            let b = canonical_type_to_typer_type(b, var_map, counter)?;
200            let c: Option<Type> = match c.as_ref() {
201                None => None,
202                Some(t) => Some(canonical_type_to_typer_type(t, var_map, counter)?),
203            };
204            Some(Type::Tuple(Box::new(a), Box::new(b), c.map(Box::new)))
205        }
206        canonical::Type::Type(name, args) => {
207            let converted: Option<Vec<Type>> = args
208                .iter()
209                .map(|a| canonical_type_to_typer_type(a, var_map, counter))
210                .collect();
211            Some(Type::Adt(name.0.clone(), converted?))
212        }
213    }
214}

216/// Convert a canonical expression to a Term.
217/// Returns None for constructs the inference engine doesn't yet handle
218/// (VarKernel, VarForeign, complex patterns inside Case).
219fn canonical_expr_to_term(
220    expr: &canonical::Expression,
221    module_types: &HashMap<Name, canonical::UnionType>,
222    counter: &mut u32,
223) -> Option<Term> {
224    match expr {
225        canonical::Expression::Int(i) => Some(Term::Int(*i as u32)),
226        canonical::Expression::Bool(b) => Some(Term::Bool(*b)),
227        canonical::Expression::Char(c) => Some(Term::Char(*c)),
228        canonical::Expression::Float(f) => Some(Term::Float(*f)),
229        canonical::Expression::VarLocal(name) => Some(Term::Identifier(name.0.clone())),
230        canonical::Expression::VarTopLevel(qname) => {
231            Some(Term::Identifier(qname.to_name().0.clone()))
232        }
233        canonical::Expression::Apply(f, a) => {
234            let fun = canonical_expr_to_term(f, module_types, counter)?;
235            let arg = canonical_expr_to_term(a, module_types, counter)?;
236            Some(Term::Apply {
237                fun: Box::new(fun),
238                arg: Box::new(arg),
239            })
240        }
241        canonical::Expression::If(cond, t, f) => {
242            let cond = canonical_expr_to_term(cond, module_types, counter)?;
243            let t = canonical_expr_to_term(t, module_types, counter)?;
244            let f = canonical_expr_to_term(f, module_types, counter)?;
245            Some(Term::If {
246                cond: Box::new(cond),
247                true_branch: Box::new(t),
248                false_branch: Box::new(f),
249            })
250        }
251        canonical::Expression::Tuple(a, b, c) => {
252            let a = canonical_expr_to_term(a, module_types, counter)?;
253            let b = canonical_expr_to_term(b, module_types, counter)?;
254            let c: Option<Term> = match c.as_ref() {
255                None => None,
256                Some(e) => Some(canonical_expr_to_term(e, module_types, counter)?),
257            };
258            Some(Term::Tuple(Box::new(a), Box::new(b), c.map(Box::new)))
259        }
260        canonical::Expression::Case(scrutinee_expr, branches) => {
261            let scrutinee = canonical_expr_to_term(scrutinee_expr, module_types, counter)?;
262            let term_branches: Vec<(TermPattern, Box<Term>)> = branches
263                .iter()
264                .map(|cb| {
265                    let (pattern, _bindings) =
266                        translate_pattern(&cb.pattern, module_types, counter)?;
267                    let body = canonical_expr_to_term(&cb.expression, module_types, counter)?;
268                    Some((pattern, Box::new(body)))
269                })
270                .collect::<Option<Vec<_>>>()?;
271            Some(Term::Case {
272                scrutinee: Box::new(scrutinee),
273                branches: term_branches,
274            })
275        }
276        // Constructors are resolved as identifiers looked up in the global env.
277        canonical::Expression::VarConstructor(qname, _) => {
278            Some(Term::Identifier(qname.to_name().0.clone()))
279        }
280        // VarForeign: not in the module's global env, skip
281        canonical::Expression::VarForeign(_, _) => None,
282        // Not yet supported: VarKernel
283        _ => None,
284    }
285}

287/// Translate a canonical pattern into a `TermPattern` plus any variable bindings
288/// introduced by the pattern.  Returns `None` for unsupported pattern shapes.
289fn translate_pattern(
290    pattern: &canonical::Pattern,
291    module_types: &HashMap<Name, canonical::UnionType>,
292    counter: &mut u32,
293) -> Option<(TermPattern, Vec<(String, Type)>)> {
294    match pattern {
295        canonical::Pattern::Anything => Some((TermPattern::Anything, vec![])),
296        canonical::Pattern::Variable(name) => {
297            // The binding's actual type will be unified with the scrutinee type in annotate.
298            Some((TermPattern::Bind(name.0.clone()), vec![]))
299        }
300        canonical::Pattern::Bool(_) => Some((
301            TermPattern::Literal(Type::Literal(TypeLiteral::Bool)),
302            vec![],
303        )),
304        canonical::Pattern::Int(_) => Some((
305            TermPattern::Literal(Type::Literal(TypeLiteral::Int)),
306            vec![],
307        )),
308        canonical::Pattern::Char(_) => Some((
309            TermPattern::Literal(Type::Literal(TypeLiteral::Char)),
310            vec![],
311        )),
312        canonical::Pattern::Constructor { ctor, args } => {
313            // Look up the parent union type to get its type variables.
314            let union_type = module_types.get(&ctor.tpe)?;
315
316            // Create fresh type vars for each ADT type parameter.
317            let mut adt_var_map: HashMap<String, TypeVariable> = HashMap::new();
318            for tv_name in &union_type.variables {
319                *counter += 1;
320                adt_var_map.insert(tv_name.0.clone(), TypeVariable { id: *counter });
321            }
322
323            // Build the ADT result type args from the fresh vars.
324            let adt_args: Vec<Type> = union_type
325                .variables
326                .iter()
327                .map(|v| Type::Variable(adt_var_map[&v.0].clone()))
328                .collect();
329
330            // Translate each constructor type parameter (reuses the same fresh vars).
331            let param_types: Vec<Type> = ctor
332                .type_parameters
333                .iter()
334                .filter_map(|t| canonical_type_to_typer_type(t, &mut adt_var_map, counter))
335                .collect();
336            if param_types.len() != ctor.type_parameters.len() {
337                return None;
338            }
339
340            // Build bindings from arg patterns.
341            let mut bindings: Vec<(String, Type)> = vec![];
342            for (arg_pattern, param_type) in args.iter().zip(param_types.iter()) {
343                match arg_pattern {
344                    canonical::Pattern::Variable(name) => {
345                        bindings.push((name.0.clone(), param_type.clone()));
346                    }
347                    canonical::Pattern::Anything => {} // no binding needed
348                    _ => return None,                  // nested complex patterns not yet supported
349                }
350            }
351
352            let term_pattern = TermPattern::Constructor {
353                adt_name: ctor.tpe.0.clone(),
354                adt_args,
355                bindings: bindings.clone(),
356            };
357            Some((term_pattern, bindings))
358        }
359        _ => None, // Tuple, Float patterns — not yet supported
360    }
361}

363/// Convert a canonical Value into a (Term, optional annotation) pair.
364/// The body is wrapped in nested Fun nodes for each pattern parameter.
365/// Returns None if any part of the value cannot be translated.
366fn value_to_term_and_annotation(
367    value: &canonical::Value,
368    module_types: &HashMap<Name, canonical::UnionType>,
369    counter: &mut u32,
370) -> Option<(Term, Option<Type>)> {
371    match value {
372        canonical::Value::Value { patterns, body, .. } => {
373            let body_term = canonical_expr_to_term(body, module_types, counter)?;
374            let term = wrap_with_patterns(patterns.iter(), body_term)?;
375            Some((term, None))
376        }
377        canonical::Value::TypedValue {
378            patterns,
379            body,
380            tpe,
381            ..
382        } => {
383            let body_term = canonical_expr_to_term(body, module_types, counter)?;
384            let pattern_iter = patterns.iter().map(|(p, _)| p);
385            let term = wrap_with_patterns(pattern_iter, body_term)?;
386            let mut var_map = HashMap::new();
387            let annotation = canonical_type_to_typer_type(tpe, &mut var_map, counter);
388            Some((term, annotation))
389        }
390    }
391}

393/// Wrap a body Term in nested Fun nodes for each pattern, outermost first.
394/// Returns None if any pattern is not translatable (e.g. constructor patterns).
395fn wrap_with_patterns<'a>(
396    patterns: impl Iterator<Item = &'a canonical::Pattern>,
397    body: Term,
398) -> Option<Term> {
399    let names: Vec<String> = patterns
400        .map(|p| match p {
401            canonical::Pattern::Variable(name) => Some(name.0.clone()),
402            canonical::Pattern::Anything => Some("_".to_string()),
403            _ => None,
404        })
405        .collect::<Option<Vec<_>>>()?;
406
407    let term = names.iter().rev().fold(body, |acc, param| Term::Fun {
408        param: param.clone(),
409        body: Box::new(acc),
410    });
411
412    Some(term)
413}

421/// Simplified pattern used inside the typer's Term.
422#[derive(Debug, Clone)]
423pub enum TermPattern {
424    /// Matches anything without binding.
425    Anything,
426    /// Binds the scrutinee type to this name.
427    Bind(String),
428    /// Matches a specific literal type; constrains the scrutinee to that type.
429    Literal(Type),
430    /// Matches an ADT constructor; carries the fresh ADT args and field bindings.
431    Constructor {
432        adt_name: String,
433        adt_args: Vec<Type>,
434        /// `(variable_name, its_type_var)` for each bound constructor argument.
435        bindings: Vec<(String, Type)>,
436    },
437}

493#[derive(Clone, Hash, PartialEq, Eq)]
494pub enum Type {
495    Literal(TypeLiteral),
496    /// A numeric literal type: unifies with both `Int` and `Float` but not other types.
497    /// This models Elm's `number` constraint for integer literals used in numeric contexts.
498    Number,
499    Variable(TypeVariable),
500    Fun {
501        param_tpe: Box<Type>,
502        return_tpe: Box<Type>,
503    },
504    Tuple(Box<Type>, Box<Type>, Option<Box<Type>>),
505    /// A named algebraic data type, e.g. `Maybe Int` → `Adt("Maybe", [Literal(Int)])`.
506    Adt(String, Vec<Type>),
507}
508
509impl std::fmt::Debug for Type {
510    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
511        match self {
512            Type::Literal(lit) => write!(f, "Lit({:?})", lit),
513            Type::Number => write!(f, "Number"),
514            Type::Variable(TypeVariable { id }) => write!(f, "Var(#{})", id),
515            Type::Fun {
516                param_tpe,
517                return_tpe,
518            } => write!(f, "Fun({:?} -> {:?})", param_tpe, return_tpe),
519            Type::Tuple(a, b, None) => write!(f, "({:?}, {:?})", a, b),
520            Type::Tuple(a, b, Some(c)) => write!(f, "({:?}, {:?}, {:?})", a, b, c),
521            Type::Adt(name, args) if args.is_empty() => write!(f, "{}", name),
522            Type::Adt(name, args) => write!(f, "{}({:?})", name, args),
523        }
524    }
525}

587        body: Box<TypedTerm>,
588    },

590        tpe: Type,

592        second: Box<TypedTerm>,

608            TypedTerm::Float { tpe, .. } => tpe,

610            TypedTerm::Fun { tpe, .. } => tpe,
611            TypedTerm::Apply { tpe, .. } => tpe,
612            TypedTerm::If { tpe, .. } => tpe,
613            TypedTerm::Let { tpe, .. } => tpe,
614            TypedTerm::Tuple { tpe, .. } => tpe,
615            TypedTerm::Case { tpe, .. } => tpe,

630
631    fn empty() -> Substitution {
632        Substitution {
633            solutions: HashMap::new(),
634        }
635    }
636
637    fn one(tvar: TypeVariable, tpe: Type) -> Substitution {
638        let mut sub = Substitution::empty();
639
640        sub.solutions.insert(tvar, tpe);
641
642        sub
643    }
644
645    // methods
646
647    fn apply(&self, c: &Constraint) -> Constraint {
648        Constraint(self.apply_type(&c.0), self.apply_type(&c.1))
649    }
650
651    fn apply_type(&self, tpe: &Type) -> Type {
652        self.solutions
653            .iter()
654            .fold(tpe.clone(), |tpe, (tvar, solution_tpe)| {
655                Substitution::substitute(tpe, tvar, solution_tpe)
656            })
657    }
658
659    fn substitute(tpe: Type, tvar: &TypeVariable, replacement: &Type) -> Type {
660        match tpe {
661            Type::Literal(_) | Type::Number => tpe,
662            Type::Fun {
663                param_tpe,
664                return_tpe,
665            } => Type::Fun {
666                param_tpe: Box::new(Substitution::substitute(*param_tpe, tvar, replacement)),
667                return_tpe: Box::new(Substitution::substitute(*return_tpe, tvar, replacement)),
668            },
669            Type::Tuple(a, b, c) => Type::Tuple(
670                Box::new(Substitution::substitute(*a, tvar, replacement)),
671                Box::new(Substitution::substitute(*b, tvar, replacement)),
672                c.map(|t| Box::new(Substitution::substitute(*t, tvar, replacement))),
673            ),
674            Type::Adt(name, args) => Type::Adt(
675                name,
676                args.into_iter()
677                    .map(|a| Substitution::substitute(a, tvar, replacement))
678                    .collect(),
679            ),
680            Type::Variable(tvar2) if tvar == &tvar2 => replacement.clone(),
681            tpe @ Type::Variable(_) => tpe,
682        }
683    }
684
685    fn merge(&self, other: Substitution) -> Substitution {
686        // This merge means we should try sub_tail first, and then sub_head
687        // Merging other in self means we apply `other` substitution to `self` solutions
688        // When merging, we want `other` solutions to take precedences over `self` solutions
689
690        let self_solutions = self
691            .solutions
692            .iter()
693            .map(|(k, v)| (k.clone(), other.apply_type(v)));
694
695        let mut sub = Substitution::empty();
696
697        sub.solutions.extend(self_solutions);
698        sub.solutions.extend(other.solutions);
699
700        sub
701    }
702}

741/// infer the type of the given term given known function defined in the outer scopes.
742/// This is a translation of the algorithm demonstrated by
743/// [Ionut Gan at I T.A.K.E Unconference 2015](https://www.youtube.com/watch?v=oPVTNxiMcSU)
744pub fn infer(term: Term, global: HashMap<String, Type>) -> Result<Type, Error> {
745    let mut env = Types::new();
746    env.extends_with(global);
747
748    let typed_term = annotate::annotate(term, &mut env)?;
749    debug!("typed term: {:#?}", typed_term);
750
751    let constraints = constraint::collect(&typed_term);
752    debug!("Constraints: {:#?}", constraints);
753
754    let substitution = unifier::unify(constraints)?;
755
756    Ok(substitution.apply_type(typed_term.tpe()))
757}

804    #[derive(Default)]
805    struct Signature {
806        counter: u8, // max 255 letters
807        known: HashMap<u32, String>,
808    }
809
810    impl Signature {
811        // Helper function to reduce boilerplate
812        fn of_type(tpe: Type) -> String {
813            let mut sig: Signature = Default::default();
814            sig.from_type(tpe)
815        }
816
817        fn from_type(&mut self, tpe: Type) -> String {
818            match tpe {
819                Type::Literal(TypeLiteral::Bool) => "Bool".to_owned(),
820                Type::Literal(TypeLiteral::Int) => "Int".to_owned(),
821                Type::Literal(TypeLiteral::Char) => "Char".to_owned(),
822                Type::Literal(TypeLiteral::Float) => "Float".to_owned(),
823                Type::Number => "number".to_owned(),
824                Type::Variable(TypeVariable { id }) => {
825                    if let Some(name) = self.known.get(&id) {
826                        name.clone()
827                    } else {
828                        let name = self.counter_as_letter();
829                        self.counter += 1;
830
831                        self.known.insert(id, name.clone());
832
833                        name
834                    }
835                }
836                Type::Fun {
837                    param_tpe,
838                    return_tpe,
839                } => {
840                    let is_param_fun = matches!(param_tpe.as_ref(), Type::Fun { .. });
841                    let param = self.from_type(*param_tpe);
842                    let retur = self.from_type(*return_tpe);
843
844                    if is_param_fun {
845                        format!("({}) -> {}", param, retur)
846                    } else {
847                        format!("{} -> {}", param, retur)
848                    }
849                }
850                Type::Tuple(a, b, None) => {
851                    format!("({}, {})", self.from_type(*a), self.from_type(*b))
852                }
853                Type::Tuple(a, b, Some(c)) => {
854                    format!(
855                        "({}, {}, {})",
856                        self.from_type(*a),
857                        self.from_type(*b),
858                        self.from_type(*c)
859                    )
860                }
861                Type::Adt(name, args) if args.is_empty() => name,
862                Type::Adt(name, args) => {
863                    let arg_strs: Vec<String> =
864                        args.into_iter().map(|a| self.from_type(a)).collect();
865                    format!("{} {}", name, arg_strs.join(" "))
866                }
867            }
868        }
869
870        fn counter_as_letter(&self) -> String {
871            let m = self.counter % 26;
872            let d = self.counter / 26;
873
874            let m_char = (97 + m) as char; // 97 is 'a'
875            let d_char = (96 + d) as char; // -1 because we start at 1
876
877            if d > 0 {
878                format!("{}{}", d_char, m_char)
879            } else {
880                format!("{}", m_char)
881            }
882        }
883    }
