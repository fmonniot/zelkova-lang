1use super::{Error, Term, TermPattern, TypeBinder, TypedTerm, Types};

3pub(super) fn annotate(term: Term, types: &mut Types) -> Result<TypedTerm, Error> {
4    match term {
5        Term::Int(value) => Ok(TypedTerm::Int {
6            tpe: types.fresh_var(),
7            value,
8        }),
9        Term::Bool(value) => Ok(TypedTerm::Bool {
10            tpe: types.fresh_var(),
11            value,
12        }),
13        Term::Char(value) => Ok(TypedTerm::Char {
14            tpe: types.fresh_var(),
15            value,
16        }),
17        Term::Float(value) => Ok(TypedTerm::Float {
18            tpe: types.fresh_var(),
19            value,
20        }),
21        Term::Fun { param, body } => {
22            let param = TypeBinder::new(param, types.fresh_var());
23            types.add_binder(param.clone());
24
25            let body = annotate(*body, types)?;
26
27            Ok(TypedTerm::Fun {
28                tpe: types.fresh_var(),
29                param,
30                body: Box::new(body),
31            })
32        }
33        Term::Identifier(name) => match types.by_name(&name) {
34            None => Err(Error::UnboundVariable(name)),
35            Some(tpe) => Ok(TypedTerm::Identifier { tpe, name }),
36        },
37        Term::Apply { fun, arg } => {
38            let fun = Box::new(annotate(*fun, types)?);
39            let arg = Box::new(annotate(*arg, types)?);
40
41            Ok(TypedTerm::Apply {
42                tpe: types.fresh_var(),
43                fun,
44                arg,
45            })
46        }
47        Term::If {
48            cond,
49            true_branch,
50            false_branch,
51        } => {
52            let cond = Box::new(annotate(*cond, types)?);
53            let true_branch = Box::new(annotate(*true_branch, types)?);
54            let false_branch = Box::new(annotate(*false_branch, types)?);
55
56            Ok(TypedTerm::If {
57                tpe: types.fresh_var(),
58                cond,
59                true_branch,
60                false_branch,
61            })
62        }
63        Term::Let {
64            binding,
65            value,
66            body,
67        } => {
68            let binding_tpe = types.fresh_var();
69            let binding = TypeBinder::new(binding, binding_tpe);
70            let value = Box::new(annotate(*value, types)?);
71
72            // scoping: We need to add the binding before evaluating the body but after the value
73            types.add_binder(binding.clone());
74            let body = Box::new(annotate(*body, types)?);
75
76            Ok(TypedTerm::Let {
77                tpe: types.fresh_var(),
78                binding,
79                value,
80                body,
81            })
82        }
83        Term::Tuple(a, b, c) => {
84            let first = Box::new(annotate(*a, types)?);
85            let second = Box::new(annotate(*b, types)?);
86            let third = c.map(|t| annotate(*t, types)).transpose()?.map(Box::new);
87            Ok(TypedTerm::Tuple {
88                tpe: types.fresh_var(),
89                first,
90                second,
91                third,
92            })
93        }
94        Term::Case {
95            scrutinee,
96            branches,
97        } => {
98            let scrutinee = Box::new(annotate(*scrutinee, types)?);
99            let mut typed_branches = Vec::new();
100            for (pattern, body) in branches {
101                // Determine what bindings this pattern introduces.
102                let new_bindings: Vec<(String, super::Type)> = match &pattern {
103                    TermPattern::Bind(name) => {
104                        // Bind to the scrutinee's type variable.
105                        vec![(name.clone(), scrutinee.tpe().clone())]
106                    }
107                    TermPattern::Constructor { bindings, .. } => bindings.clone(),
108                    TermPattern::Anything | TermPattern::Literal(_) => vec![],
109                };
110                for (name, tpe) in &new_bindings {
111                    types.add_binder(TypeBinder::new(name.clone(), tpe.clone()));
112                }
113                let typed_body = Box::new(annotate(*body, types)?);
114                // Restore scope: remove bindings added for this branch.
115                for (name, _) in &new_bindings {
116                    types.remove_binder(name);
117                }
118                typed_branches.push((pattern, typed_body));
119            }
120            Ok(TypedTerm::Case {
121                tpe: types.fresh_var(),
122                scrutinee,
123                branches: typed_branches,
124            })
125        }
126    }
127}
