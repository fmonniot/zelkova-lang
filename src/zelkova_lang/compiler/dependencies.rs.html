1//! Walk package's module dependencies
2//!
3//! Within a package, each `Module` will depends on other `Module`s. The aim
4//! of the dependencies module is to provide a way to sort the modules in a dependent
5//! way, allowing parallel action on which module. We will also find out existing
6//! cyclic dependencies and return all of them as error.
7//!
8//! We are currently using the excellent `petgraph` crate for sorting our dependencies
9//! and discovering cycle. We currently use the results of the cycle discovery to drive
10//! how we execute the processing. If we want to introduce parallel processing of the
11//! different module, we will have to be a bit more thorough because we currently don't
12//! have any information _at processing time_ to know when to block the check of some
13//! modules. We don't know how we will do most of that processing, so it's a bit
14//! premature to actually implement multi threading here.
15//! Let's see how it goes first on a single thread and we can revisit this decision
16//! later on.
17//!
18

27// TODO Improve this with barrier on each module, to be able to parallelize
28// the processing down thel line.
29pub struct ModuleWalker<'a> {
30    /// The list of modules to process. We loose the barrier orders
31    /// so we can't really process them in parallel at the moment.
32    modules: Vec<&'a Module>,
33}
34
35impl<'a> std::fmt::Debug for ModuleWalker<'a> {
36    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
37        let modules: Vec<_> = self.modules.iter().map(|m| &m.name).collect();
38
39        f.debug_struct("ModuleWalker")
40            .field("modules", &modules)
41            .finish()
42    }
43}
44
45#[derive(Debug, PartialEq)]
46pub enum Error {
47    CycleDetected(Vec<Vec<Name>>),
48}
49
50impl<'a> ModuleWalker<'a> {
51    pub fn new(modules: &'a Vec<Module>) -> Result<ModuleWalker, Error> {
52        let mut graph = DiGraph::new();
53
54        let mut names = HashMap::new();
55
56        // First populate the graph with all nodes, and keep a reference from Path to the node index
57        for module in modules.iter() {
58            let idx = graph.add_node(module);
59            names.insert(&module.name, idx);
60        }
61
62        // Now that all modules have been inserted, we can create edges
63        for module in modules.iter() {
64            let m_idx = names.get(&module.name).unwrap();
65
66            // When building the dependency graph, we exclude all imports for
67            // modules not in this package. The reason is simple: a dependent
68            // package cannot depends on the current package, so no cycle to detect
69            for dep in module
70                .imports
71                .iter()
72                .filter_map(|import| names.get(&import.name))
73            {
74                graph.add_edge(*m_idx, *dep, ());
75            }
76        }
77
78        // Find the strongly connected graphs (scc), if there are more than one node per scc
79        // it means there is a circular dependency.
80        let scc = petgraph::algo::tarjan_scc(&graph);
81
82        let (cycles, deps): (Vec<_>, Vec<_>) = scc.iter().partition(|&v| v.len() > 1);
83
84        if cycles.is_empty() {
85            let modules = deps.into_iter().flatten().map(|&idx| graph[idx]).collect();
86
87            Ok(ModuleWalker { modules })
88        } else {
89            let c = cycles
90                .into_iter()
91                .map(|cycle| {
92                    cycle
93                        .into_iter()
94                        .map(|&idx| graph[idx].name.clone())
95                        .collect()
96                })
97                .collect();
98
99            Err(Error::CycleDetected(c))
100        }
101    }
102
103    /// Given a package name, a set of existing interfaces and a checker function,
104    /// Check each module in its dependencies order
105    pub fn check_in_order<E>(
106        &self,
107        package: &crate::compiler::PackageName,
108        interfaces: &mut HashMap<Name, crate::compiler::Interface>,
109        check: fn(
110            package: &crate::compiler::PackageName,
111            interfaces: &HashMap<Name, crate::compiler::Interface>,
112            source: &crate::compiler::parser::Module,
113        ) -> Result<super::canonical::Module, E>,
114    ) -> Result<Vec<crate::compiler::canonical::Module>, Vec<E>> {
115        let iter = self
116            .modules
117            .iter()
118            .map(|module| match check(package, interfaces, module) {
119                Ok(m) => {
120                    // Once we have successfuly checked a module, we can add it to the available interfaces
121                    // for the following modules.
122                    // TODO We might want to have a less strict approach if we want to make some progress
123                    // in dependent modules even if the current one doesn't pass all checks (accumulate
124                    // more errors to show at once to the programmer).
125                    // In term of types, it means having check return something like Result<(Module, Errors), Errors>.
126                    // The overall Result is for the errors stopping us from building the minimum required to make
127                    // a Module. This might not be required if we decide how to default the Exports (the rest is vec based,
128                    // so can be default to ignore). This also means the whole canonicalization process will have to be
129                    // rewritten to have scoped-fail semantics.
130                    let iface_name = m.name.name().clone();
131                    let iface = m.to_interface();
132                    debug!("Inserting {} with value {:?}", iface_name, iface);
133                    interfaces.insert(iface_name, iface);
134
135                    Ok(m)
136                }
137                Err(err) => Err(err),
138            });
139
140        collect_accumulate(iter)
141    }
142}

144#[cfg(test)]
145mod tests {
146    use super::*;
147    use crate::compiler::parser::{Exposing, Import};

150    fn module<S: Into<String>>(name: S, deps: Vec<S>) -> Module {
151        let imports = deps
152            .into_iter()
153            .map(|n| Import {
154                name: Name(n.into()),
155                alias: None,
156                exposing: Exposing::Open,
157            })
158            .collect();
159
160        Module {
161            name: Name(name.into()),
162            binding_javascript: false,
163            exposing: Exposing::Open,
164            imports,
165            infixes: vec![],
166            types: vec![],
167            functions: vec![],
168        }
169    }
170
171    fn name<S: Into<String>>(s: S) -> Name {
172        Name(s.into())
173    }
174
175    fn dummy_check(
176        package: &PackageName,
177        _interfaces: &HashMap<Name, Interface>,
178        source: &parser::Module,
179    ) -> Result<canonical::Module, ()> {
180        Ok(canonical::Module {
181            name: ModuleName::new(package.clone(), source.name.clone()),
182            exports: canonical::Exports::Everything,
183            infixes: HashMap::new(),
184            types: HashMap::new(),
185            values: HashMap::new(),
186        })
187    }
188
189    fn assert_walker_processed_order(walker: ModuleWalker, expected: Vec<&str>) {
190        let name = crate::compiler::PackageName::new("author", "project");
191        let mut ifaces = HashMap::new();
192        let res: Result<Vec<canonical::Module>, Vec<()>> =
193            walker.check_in_order(&name, &mut ifaces, dummy_check);
194
195        assert_eq!(
196            res.map(|v| v
197                .into_iter()
198                .map(|m| m.name.name().0.clone())
199                .collect::<Vec<_>>()),
200            Ok(expected.into_iter().map(|s| s.to_string()).collect())
201        );
202    }

204    #[test]
205    fn dependencies_without_cycle() {
206        let a = module("a", vec![]);
207        let b = module("b", vec!["a"]);
208        let c = module("c", vec!["b"]);
209        let d = module("d", vec!["a"]);
210
211        let modules = vec![a, b, c, d];
212
213        let walker = ModuleWalker::new(&modules);
214
215        assert_walker_processed_order(walker.expect("no errors here"), vec!["a", "b", "c", "d"])
216    }

244    fn dependencies_with_branches() {
245        /*
246        a <- c <- b <- d
247          <- e <- f <- g <- h
248
249        i -> b
250          -> e
251          <- h
252        i should be after e, c, and b; h should be after i; no guarantee on the order when there is no constraint
253        */
254        let a = module("a", vec![]);
255        let b = module("c", vec!["a"]);
256        let c = module("b", vec!["c"]);
257
258        let d = module("d", vec!["c"]);
259        let e = module("e", vec!["a"]);
260        let f = module("f", vec!["e"]);
261        let g = module("g", vec!["f"]);
262        let h = module("h", vec!["g", "i"]);
263
264        let i = module("i", vec!["b", "e"]);
265
266        let modules = vec![a, c, b, d, e, f, g, h, i];
